// This is a true love story.
public void loveStory(List<String> girls) {
// Once there was a boy named Vincent.
String theBoy = "Vincent";
// He had no girlfriend and he was not happy.
String theGirl = null;
String status = "single";
boolean happy = false;
List<String> family;

// Until he met with a girl named Helen.

// He knew she is the girl in his life.
theGirl = aGirl;
// He couldn't help falling in love with her.
status = "in love";
// And he has been happy ever since.
happy = true;
break;
}
}

// This is how the love story about the boy and the girl begins.
// But this is not the end of the story.
// The boy and the girl will live together happily since then.
for(int yearTogether = 0; yearTogether <= Integer.MAX_VALUE; yearTogether++) {
// After two years together, they will be married.
if(yearTogether == 2) {
status = "married";
// They will form a new family together.
family = new ArrayList<String>();
family.add(theBoy);
family.add(theGirl);
}
// After four years together, their family will have a new member.
if(yearTogether == 4) {
family.add("baby");
}
// Nothing but the end of the world can set them apart.
}

System.out.println("Baby, together let's make the the story come true.");
}
